import librosa

def analyze_audio(file_path):
    y, sr = librosa.load(file_path)

    tempo, _ = librosa.beat.beat_track(y=y, sr=sr)
    spectral_centroid = librosa.feature.spectral_centroid(y=y, sr=sr).mean()

    return {
        "tempo": tempo,
        "energy": spectral_centroid
    }
